This is a great shift in direction. Moving from a "fly-over" style to a **Travel-based Exploration** system turns the map into a strategic tool rather than just a menu.

To achieve this, we will remove the location pins from the map for "unvisited" or "enemy" areas. The user will only see these on the sidebar. When they click "Travel," the tokens on the map will physically move toward that coordinate.

### 2. The "Travel" Mechanic (Visual Movement)

Instead of `map.flyTo`, we will animate the markers. We can use a simple `setInterval` or a CSS transition to move the character tokens toward the target location.

```javascript
function startTravel(locationId) {
    if (campaignState.isTraveling) return;
    
    const loc = locationStorage.get(locationId).data;
    const targetLatLng = [loc.lat, loc.lng];
    
    campaignState.isTraveling = true;
    
    // Find our main party tokens (e.g., Washington)
    const partyTokens = currentTokens.filter(t => t.side === 'Continental');

    // Visual cue: Change sidebar to "In Transit"
    document.body.classList.add('traveling');

    // Simulate movement: move markers every 100ms
    let step = 0;
    const totalSteps = 50;
    
    const moveInterval = setInterval(() => {
        step++;
        partyTokens.forEach(token => {
            // Simple linear interpolation for demonstration
            const currentLat = token.gps.lat + (loc.lat - token.gps.lat) * (step / totalSteps);
            const currentLng = token.gps.lng + (loc.lng - token.gps.lng) * (step / totalSteps);
            
            // Update token data
            token.gps.lat = currentLat;
            token.gps.lng = currentLng;
            
            // Check for Random Encounter (2% chance per step)
            if (Math.random() < 0.02) {
                clearInterval(moveInterval);
                triggerEncounter();
            }
        });
        
        renderWorldMarkers(); // Re-draw markers at new positions
        
        if (step >= totalSteps) {
            clearInterval(moveInterval);
            campaignState.isTraveling = false;
            document.body.classList.remove('traveling');
            enterTacticalView(locationStorage.get(locationId));
        }
    }, 100);
}

```

### 3. Random Encounter Popups

This adds the "threat" element you mentioned. When travel is interrupted, we throw a modal over the screen.

```javascript
function triggerEncounter() {
    const encounterOverlay = document.createElement('div');
    encounterOverlay.className = 'encounter-modal';
    encounterOverlay.innerHTML = `
        <div class="encounter-content">
            <h1>‚ö†Ô∏è AMBUSH!</h1>

            <button onclick="resolveEncounter('fight')">Battle Stations</button>
            <button onclick="resolveEncounter('flee')">Attempt to Retreat</button>
        </div>
    `;
    document.body.appendChild(encounterOverlay);
}

```
