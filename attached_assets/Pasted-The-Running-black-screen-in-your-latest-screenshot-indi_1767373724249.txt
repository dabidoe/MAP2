### 2. Robust Initialization (The "Black Screen" Fix)

The black screen often happens because `fetch()` fails. We will add a fallback so the app loads even if a JSON file is missing.

**Update `initCampaign` in `script.js`:**

```javascript
async function initCampaign() {
    try {
        console.log("Initializing War Room...");
        
        // Load Locations with fallback
        let locations = [];
        try {
            const locRes = await fetch('/data/locations.json');
            locations = await locRes.json();
        } catch (e) {
            console.error("Location fetch failed, using defaults.");
            locations = [
                { id: 'frozen_vigil', title: 'McConkey\'s Ferry', lat: 40.2985, lng: -74.8718, description: 'The embarkation point on the Delaware.' },
                { id: 'trenton_barracks', title: 'Trenton Barracks', lat: 40.2178, lng: -74.7681, description: 'The Hessian stronghold.' }
            ];
        }

        // Load Characters from your uploaded characters.json
        try {
            const charRes = await fetch('/data/characters.json');
            const charData = await charRes.json();
            
            // Map side and starting position to your characters
            currentTokens = charData.map(c => ({
                ...c,
                side: (c.name.includes('Washington') || c.name.includes('Greene')) ? 'Continental' : 'Hessian',
                gps: { lat: 40.2985, lng: -74.8718 } // Start at the ferry
            }));
        } catch (e) {
            console.error("Character fetch failed.");
        }

        // Setup storage
        locations.forEach(loc => locationStorage.set(loc.id, { data: loc }));

        // Final Render
        renderWorldSidebar();
        renderWorldMarkers();
        
        console.log("Systems Online.");
    } catch (criticalError) {
        console.error("CRITICAL BOOT ERROR:", criticalError);
        document.body.innerHTML = `<div style="color:white; padding:20px;">Boot Error: ${criticalError.message}</div>`;
    }
}

```

### 4. Travel Feature Logic

When you click **"Command Travel"**, the Continental markers will move toward the destination.

```javascript
function startTravel(targetId) {
    const destination = locationStorage.get(targetId).data;
    
    // Filter for your Continental party members like George Washington
    const party = currentTokens.filter(t => t.side === 'Continental');
    
    party.forEach(token => {
        // We'll use Leaflet's flyTo for the camera, 
        // but we update the token's internal GPS for the logic
        token.gps.lat = destination.lat;
        token.gps.lng = destination.lng;
    });

    map.flyTo([destination.lat, destination.lng], 14, { duration: 3 });
    
    // Encounter Check: 20% chance of ambush during travel
    if (Math.random() < 0.20) {
        setTimeout(() => triggerEncounterPopup(), 1500);
    }

    renderWorldMarkers();
}

```
